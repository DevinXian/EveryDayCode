给数组对象添加方法(面向对象)

dyygtfx Updated at Mar 01, 2015

 Array.prototype.sum = function () {
            for (var sum = i = 0;i < this.length; i++) {
                sum += parseInt(this[i]);
            }
            return sum;
        }
        Array.prototype.maxima = function () {
            for (var i = 0, maxValue = Number.MIN_VALUE; i < this.length; i++) {
                parseInt(this[i]) > maxValue && (maxValue = this[i]);
            }
            return maxValue;
        }
        var arr = new array;
        console.log(arr.join("+") + "=" + arr.sum());
        console.log(arr.join("|") + "中，最大的数是：" + arr.maxima());
