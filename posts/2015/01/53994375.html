#js# 文本框自动伸缩

dyygtfx Updated at Feb 11, 2015

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>自动伸缩</title>
</head>
<body>
    <textarea name="" id="txtContent" cols="30" rows="10" oninput="ResizeTextarea()" onkeyup="ResizeTextarea()">
        hhhhhhhhhhhhhhhhhh
    </textarea>
    <script>
        var minHeight = 200;
        var maxHeight = 400;
        function ResizeTextarea() {
            var t = document.getElementById('txtContent');
            h = t.scrollHeight;
            if (maxHeight > h > minHeight) {
                t.style.height = h + "px";
            } else if (h > maxHeight) {
                t.style.height = maxHeight;
            } else {
                t.style.height = h + "px";
            }
        }
    </script>
</body>
</html>
