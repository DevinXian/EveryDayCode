JS 求中位数

dyygtfx Updated at Feb 16, 2016

function getMedian(arr) {
    var len = arr.length;
    arr.sort(function (a,b) {
        return (a-b);
    });
    var lowMiddle = Math.floor((len - 1) / 2);
    var hightMiddle = Math.ceil((len - 1) / 2);
    var median = (arr[lowMiddle] + arr[hightMiddle]) / 2;
    return median;
}
