js 16进制颜色值转 RGB

dyygtfx Updated at Oct 10, 2016

const toRGB = (() => {
  const expand = hex =>
    hex.length < 7 ? hex.split("").reduce((a, b) => a + b + b) : hex;
  const convert = hex =>
    hex.match(/[\d\w]{2}/g).map(val => parseInt(val, 16));
  return hex => {
    const [r, g, b] = convert(expand(hex));
    return `rgb(${r}, ${g}, ${b})`;
  };
})();
