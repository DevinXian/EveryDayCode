点击DIV就闪烁的效果

dyygtfx Updated at Feb 11, 2015

<!doctype html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>div闪烁</title>
    <style>

    #box {
            position:absolute;
            top:50%;
            left:50%;
            color:#fff;
            width:200px;
            height:200px;
            background:red;
            cursor:pointer;
            letter-spacing:5px;
            text-align:center;
            font:12px/200px Arial;margin:-100px 0 0 -100px;
        }

    </style>
    <script>
        window.onload = function() {
            var oBox = document.getElementById("box");
            var timer = null;
            oBox.onclick = function() {
                var i = 0;
                clearInterval(timer);
                timer = setInterval(function() {
                    oBox.style.display = i++%2?"none":"block";
                    i > 6&&(clearInterval(timer))
                },80)
            }
        }
    </script>
</head>
<body>
    <div id="box">点击我就闪</div>
</body>
</html>
